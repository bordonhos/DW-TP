-- PVP Total
select	measures.pvp on columns 
from	ed77233

-- PVP por distrito
select
	(	
		measures.pvp,
		[dim clientes].distrito.distrito
	)
on columns
from	ed77233

-- Aveiro
select
	(	
		measures.pvp
	)
on columns
from	ed77233
where	[dim clientes].distrito.&[Aveiro]

-- PVP por distrito em 2014
select
	(	
		measures.pvp,
		[dim clientes].distrito.distrito
	)
on columns
from	ed77233
where	[dim tempo].Ano.&[2014]

-- PVP por trimestre
select 
	(
		measures.pvp,
		[dim tempo].trimestre.trimestre
	)
on columns
from	ed77233

-- PVP por faixa etaria
select
	(	
		measures.pvp,
		[DD_Vendas].[Faixa etaria].[faixa etaria]
	)
on columns
from	ed77233

-- Nº de vendas por faixa etaria
select
	(	
		measures.[fact vendas count],
		[DD_Vendas].[Faixa etaria].[faixa etaria]
	)
on columns
from	ed77233

-- PVP por estação
select
	(	
		measures.pvp,
		[DD_Vendas].[estacao].[estacao]
	)
on columns
from	ed77233

-- Nº de vendas por estação
select
	(	
		measures.[fact vendas count],
		[DD_Vendas].[estacao].[estacao]
	)
on columns
from	ed77233

select
	(
		[measures].[PVP],
		[dim tempo].Ano.Ano
	)
on columns
from	ed77233

-- Preco Final por ano
WITH 
   MEMBER [Measures].[PrecoFinal] AS
   [Measures].[PVP] - [Measures].[Desconto]
select
	(
		[measures].[PrecoFinal],
		[dim tempo].Ano.Ano
	)
on columns
from	ed77233

WITH 
   MEMBER [Measures].[PrecoFinal] AS
   [Measures].[PVP] - [Measures].[Desconto]
select
	(
		[measures].[PrecoFinal],	
		[dim tempo].Ano.&[2014],
		[dim tempo].Trimestre.&[3],
		[dim clientes].Distrito.&[Aveiro],
		[dim produtos].Tamanho.&[S],
		[dim produtos].Marca.&[LA Women]
	)
on columns
from	ed77233

select 
	(
		measures.[per desc],
		[dim produtos].Marca.Marca
	)
on columns
from ed77233

-- Por linha

   MEMBER [Measures].[PercDesconto] AS
   [Measures].[Per Desc] / [Measures].[FACT Vendas Count]

		 [Measures].[PercDesconto] on columns,
		 [dim clientes].Distrito.Distrito on rows
from ed77233
